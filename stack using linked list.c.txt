#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node *next;
};

struct Node *top = NULL;

// Function to push element
void push(int value) {
    struct Node *newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = value;
    newNode->next = top;
    top = newNode;
}

// Function to display top element (peek)
void peek() {
    if(top == NULL)
        printf("Stack is Empty\n");
    else
        printf("Top element = %d\n", top->data);
}

int main() {
    push(10);
    push(20);
    push(30);

    peek();   // Display top element

    return 0;
}
